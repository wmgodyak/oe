namespace modules\blog\widgets;

use system\core\Widget;

defined("CPATH") or die();

/**
 * Class Sample
 * @package modules\blog\widgets
 */
class Sample extends Widget
{
    public function __construct()
    {
        parent::__construct('blog.sample', 'Sample One', 'Sample description');
    }

    /**
     * @param array $data
     * @return string
     */
    public function form($data = [])
    {
        $data['name'] = isset($data['name']) ? $data['name'] : null;
        return "
            <div class=\"form-group\">
                <label for=\"name\" class=\"col-sm-3 control-label\">Назва</label>
                <div class=\"col-sm-9\">
                    <input name=\"{$this->getFieldName('name')}\" id=\"{$this->getFieldID('name')}\" class=\"form-control\" value=\"{$data['name']}\" required>
                </div>
            </div>
        ";

    }

    public function display($args, $data)
    {
        echo 'Sample::display()';
    }
}
